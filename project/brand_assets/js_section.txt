    { code: 'LOVE15',      type: 'percent', value: 15,  label: '15% off — LOVE15'       },
    { code: 'FLAT50',      type: 'fixed',   value: 50,  label: 'QR 50 off — FLAT50'     },

  function loadCart() {
    try { return JSON.parse(localStorage.getItem(CART_KEY)) || []; }
    catch (e) { return []; }
  }

  function loadPromo() {
    try { return JSON.parse(localStorage.getItem(PROMO_KEY)) || null; }
    catch (e) { return null; }
  }

  function loadGiftMsg() {
    return localStorage.getItem(GIFT_MSG_KEY) || '';
  }

  function loadDelivery() {
    try { return JSON.parse(localStorage.getItem(DELIVERY_KEY)) || { date: '', slot: '' }; }
    catch(e) { return { date: '', slot: '' }; }
  }

  function loadRecipient() {
    try {
      return JSON.parse(localStorage.getItem(RECIPIENT_KEY)) ||
        { name:'', phone:'', address:'', landmark:'', occasion:'', notes:'' };
    } catch(e) {
      return { name:'', phone:'', address:'', landmark:'', occasion:'', notes:'' };
    }
  }

  function saveCart()      { localStorage.setItem(CART_KEY,      JSON.stringify(cart));          }
  function savePromo()     { localStorage.setItem(PROMO_KEY,     JSON.stringify(appliedPromo));  }
  function saveGiftMsg()   { localStorage.setItem(GIFT_MSG_KEY,  giftMessage);                   }
  function saveDelivery()  { localStorage.setItem(DELIVERY_KEY,  JSON.stringify(deliverySlot));  }
  function saveRecipient() { localStorage.setItem(RECIPIENT_KEY, JSON.stringify(recipient));     }

  let cart         = loadCart();

  let giftMessage  = loadGiftMsg();
  let deliverySlot = loadDelivery();

  // ── Recipient helpers ─────────────────────────
  function onRecipientInput(field, value) {
    recipient[field] = value;
    saveRecipient();
    const id = 'rf' + field.charAt(0).toUpperCase() + field.slice(1);
    const el = document.getElementById(id);
    if (el) el.classList.remove('error');
    document.getElementById('rfCheckoutError').classList.remove('show');
  }

  function validateRecipient() {
    let valid = true;
    ['rfName', 'rfPhone', 'rfAddress'].forEach(id => {
      const el = document.getElementById(id);
      if (!el || !el.value.trim()) {
        if (el) el.classList.add('error');
        valid = false;
      }
    });
    return valid;
  }

  function validateAndOrder() {
    if (!validateRecipient()) {
      const err = document.getElementById('rfCheckoutError');
      err.classList.add('show');
      setTimeout(() => err.classList.remove('show'), 4000);
      const firstErr = document.querySelector('#cartPanel .rf-input.error, #cartPanel .rf-textarea.error');
      if (firstErr) firstErr.scrollIntoView({ behavior: 'smooth', block: 'center' });
      return false;
    }
    return true;
  }

  // ── Delivery helpers ──────────────────────────
  function toDateStr(d) {
    return `${d.getFullYear()}-${String(d.getMonth()+1).padStart(2,'0')}-${String(d.getDate()).padStart(2,'0')}`;
  }

  function getMinDate() {
    const now = new Date();
    const anySlotToday = TIME_SLOTS.some(s => s.start > now.getHours() + SLOT_LEAD_HRS);
    if (!anySlotToday) {
      const tom = new Date(now); tom.setDate(tom.getDate() + 1);
      return toDateStr(tom);
    }
    return toDateStr(now);
  }

  function getAvailableSlots(dateStr) {
    const now = new Date();
    if (dateStr === toDateStr(now)) {
      return TIME_SLOTS.filter(s => s.start > now.getHours() + SLOT_LEAD_HRS);
    }
    return TIME_SLOTS;
  }

  function formatDeliveryDate(dateStr) {
    if (!dateStr) return '';
    const [y, m, d] = dateStr.split('-').map(Number);
    return new Date(y, m - 1, d).toLocaleDateString('en-GB', {
      weekday: 'short', day: 'numeric', month: 'short', year: 'numeric'
    });
  }

  function onDeliveryDateChange(val) {
    deliverySlot.date = val;
    deliverySlot.slot = '';
    saveDelivery();
    renderDeliverySlots(val);
    updateCartUI();
  }

  function selectDeliverySlot(slotId) {
    deliverySlot.slot = slotId;
    saveDelivery();
    document.querySelectorAll('.delivery-slot-btn').forEach(btn => {
      btn.classList.toggle('active', btn.dataset.slot === slotId);
    });
    updateCartUI();
  }

  function resetDeliverySlot() {
    deliverySlot.slot = '';
    saveDelivery();
    const slotsEl = document.getElementById('deliverySlots');
    if (slotsEl) {
      slotsEl.querySelectorAll('.delivery-slot-btn').forEach(b => b.classList.remove('active'));
    }
    updateCartUI();
  }

  function renderDeliverySlots(dateStr) {
    const container = document.getElementById('deliverySlots');
    const noteEl    = document.getElementById('deliveryNote');
    if (!container) return;

    if (!dateStr) { container.innerHTML = ''; noteEl.textContent = ''; return; }

    const now      = new Date();
    const slots    = getAvailableSlots(dateStr);
    const isToday  = dateStr === toDateStr(now);

    noteEl.textContent = isToday
      ? `✦ Same-day delivery · cutoff ${SAME_DAY_CUTOFF}:00`
      : '';
    noteEl.className = 'delivery-note' + (isToday ? ' same-day' : '');

    if (slots.length === 0) {
      container.innerHTML = '<p class="delivery-no-slots">No slots available today — please pick another date.</p>';
      return;
    }

    container.innerHTML = slots.map(s => `
      <button class="delivery-slot-btn${deliverySlot.slot === s.id ? ' active' : ''}"
        data-slot="${s.id}" onclick="selectDeliverySlot('${s.id}')">${s.label}</button>
    `).join('');
  }

  // ── Add to cart ───────────────────────────────
  function addToCart(btn, name) {
    const card    = btn.closest('.prod-card');
    const catEl   = card ? card.querySelector('.prod-cat')   : null;
    const priceEl = card ? card.querySelector('.prod-price') : null;

    let price = 0;
    if (priceEl) {
      const clone = priceEl.cloneNode(true);
      const cur = clone.querySelector('.currency');
      if (cur) cur.remove();
      price = parseFloat(clone.textContent.trim()) || 0;
    }

    const cat = catEl ? catEl.textContent.trim() : '';
    const existing = cart.find(i => i.name === name);
    if (existing) {
      existing.qty++;
    } else {
      cart.push({ id: Date.now(), name, cat, price, qty: 1 });
    }

    saveCart();
    updateCartUI();
    openCart();

    const orig = btn.innerHTML;
    btn.innerHTML = '✓ Added';
    btn.style.background = '#4a8c4a';
    setTimeout(() => { btn.innerHTML = orig; btn.style.background = ''; }, 1400);
  }

  // ── Promo code ────────────────────────────────
  function applyPromo() {
    const input = document.getElementById('promoInput');
    const msgEl = document.getElementById('promoMsg');
    const code  = input.value.trim().toUpperCase();

    msgEl.className = 'promo-msg';
    msgEl.textContent = '';

    if (!code) {
      msgEl.className = 'promo-msg error';
      msgEl.textContent = 'Please enter a promo code.';
      return;
    }

    if (appliedPromo) {
      msgEl.className = 'promo-msg error';
      msgEl.textContent = 'Remove the current promo before applying a new one.';
      return;
    }

    const match = PROMO_CODES.find(p => p.code === code);
    if (!match) {
      msgEl.className = 'promo-msg error';
      msgEl.textContent = 'Invalid promo code. Please try again.';
      input.value = '';
      return;
    }

    appliedPromo = match;
    savePromo();
    input.value = '';
    msgEl.className = 'promo-msg success';
    msgEl.textContent = '✓ Promo code applied!';
    updateCartUI();

    setTimeout(() => { msgEl.textContent = ''; msgEl.className = 'promo-msg'; }, 3000);
  }

  function removePromo() {
    appliedPromo = null;
    savePromo();
    document.getElementById('promoMsg').textContent = '';
    document.getElementById('promoMsg').className = 'promo-msg';
    updateCartUI();
  }

  // ── Update UI ─────────────────────────────────
  function updateCartUI() {
    const subtotal = cart.reduce((s, i) => s + i.price * i.qty, 0);
    const count    = cart.reduce((s, i) => s + i.qty, 0);

    let discount = 0;
    if (appliedPromo && subtotal > 0) {
      if (appliedPromo.type === 'percent') {
        discount = subtotal * appliedPromo.value / 100;
      } else {
        discount = Math.min(appliedPromo.value, subtotal);
      }
    }
    const total = Math.max(0, subtotal - discount);

    // Badge
    const badge = document.getElementById('cartCount');
    badge.textContent = count;
    badge.classList.toggle('visible', count > 0);

    // Footer visibility
    document.getElementById('cartFooter').style.display = cart.length ? 'block' : 'none';

    // Subtotal row — only show when a promo is applied
    const hasDiscount = appliedPromo && discount > 0;
    document.getElementById('cartSubtotalRow').style.display    = hasDiscount ? 'flex'  : 'none';
    document.getElementById('cartDiscountRow').style.display    = hasDiscount ? 'flex'  : 'none';
    document.getElementById('cartTotalsDivider').style.display  = hasDiscount ? 'block' : 'none';
    document.getElementById('cartSubtotalVal').textContent      = subtotal.toFixed(0);
    document.getElementById('cartDiscountVal').textContent      = discount.toFixed(0);
    document.getElementById('cartTotalVal').textContent         = total.toFixed(0);

    // Applied promo tag
    const appliedEl = document.getElementById('promoApplied');
    if (appliedPromo) {
      document.getElementById('promoAppliedLabel').textContent = appliedPromo.label;
      appliedEl.style.display = 'flex';
      document.getElementById('promoSection').style.display = 'none';
    } else {
      appliedEl.style.display = 'none';
      document.getElementById('promoSection').style.display = 'block';
    }

    // WhatsApp message
    let msg = 'Hello Rosebella! 🌸 I would like to order:\n\n';
    cart.forEach(i => {
      msg += `• ${i.name} x${i.qty} — QR ${(i.price * i.qty).toFixed(0)}\n`;
      if (i.variants) {
        const v = i.variants;
        const parts = [];
        if (v.size)                         parts.push(v.size);
        if (v.packaging)                    parts.push(v.packaging);
        if (v.addons && v.addons.length)    parts.push(v.addons.join(', '));
        if (parts.length) msg += `  ↳ ${parts.join(' · ')}\n`;
      }
    });
    if (appliedPromo && discount > 0) {
      msg += `\nPromo: ${appliedPromo.code} (−QR ${discount.toFixed(0)})`;
    }
    msg += `\nTotal: QR ${total.toFixed(0)}`;
    if (deliverySlot.date && deliverySlot.slot) {
      const slotObj = TIME_SLOTS.find(s => s.id === deliverySlot.slot);
      msg += `\n\n📦 Delivery: ${formatDeliveryDate(deliverySlot.date)}`;
      msg += `\n⏰ Time Slot: ${slotObj ? slotObj.label : deliverySlot.slot}`;
    }
    if (recipient.name || recipient.phone || recipient.address) {
      msg += '\n\n👤 Recipient Details:';
      if (recipient.name)     msg += `\nName: ${recipient.name}`;
      if (recipient.phone)    msg += `\nPhone: ${recipient.phone}`;
      if (recipient.address)  msg += `\nAddress: ${recipient.address}`;
      if (recipient.landmark) msg += `\nLandmark: ${recipient.landmark}`;
      if (recipient.occasion) msg += `\nOccasion: ${recipient.occasion}`;
      if (recipient.notes)    msg += `\nNotes: ${recipient.notes}`;
    }
    if (deliverySlot.date && deliverySlot.slot) {
      const slotObj = TIME_SLOTS.find(s => s.id === deliverySlot.slot);
      msg += `\n\n📦 Delivery: ${formatDeliveryDate(deliverySlot.date)}`;
      msg += `\n⏰ Time Slot: ${slotObj ? slotObj.label : deliverySlot.slot}`;
    }
    if (giftMessage.trim()) {
      msg += `\n\n💌 Gift Message:\n"${giftMessage.trim()}"`;
    }
    document.getElementById('cartWaBtn').href = 'https://wa.me/97433446618?text=' + encodeURIComponent(msg);

    // Empty state
    document.getElementById('cartEmptyState').style.display = cart.length ? 'none' : 'flex';

    // Items
    const container = document.getElementById('cartItemsContainer');
    container.innerHTML = cart.map(item => {
      const variantLine = (() => {
        if (!item.variants) return '';
        const v = item.variants;
        const parts = [];
        if (v.size)                      parts.push(v.size);
        if (v.packaging)                 parts.push(v.packaging);
        if (v.addons && v.addons.length) parts.push(...v.addons.map(a => `+${a}`));
        return parts.length ? `<div class="cart-item-variants">${parts.join(' · ')}</div>` : '';
      })();
      return `
      <div class="cart-item" id="ci-${item.id}">
        <div class="cart-item-img" style="background:var(--charcoal2);border-radius:8px;flex-shrink:0;width:72px;height:72px;"></div>
        <div class="cart-item-info">
          <span class="cart-item-cat">${item.cat}</span>
          <div class="cart-item-name">${item.name}</div>
          ${variantLine}
          <div class="cart-item-controls">
            <button class="qty-btn" onclick="changeQty(${item.id}, -1)">−</button>
            <span class="qty-num">${item.qty}</span>
            <button class="qty-btn" onclick="changeQty(${item.id}, 1)">+</button>
          </div>
        </div>
        <div style="display:flex;flex-direction:column;align-items:flex-end;gap:8px;">
          <div class="cart-item-price"><span class="ci-cur">QR</span>${(item.price * item.qty).toFixed(0)}</div>
          <button class="cart-item-del" onclick="removeItem(${item.id})" title="Remove">🗑</button>
        </div>
      </div>`;
    }).join('');

    // Recipient fields — sync on load/restore
    const rfMap = { rfName:'name', rfPhone:'phone', rfAddress:'address', rfLandmark:'landmark', rfOccasion:'occasion', rfNotes:'notes' };
    Object.entries(rfMap).forEach(([id, field]) => {
      const el = document.getElementById(id);
      if (el && el.value !== recipient[field]) el.value = recipient[field];
    });

    // Gift message — sync textarea value and counter on load/restore
    const ta = document.getElementById('giftMsgInput');
    if (ta && ta.value !== giftMessage) ta.value = giftMessage;
    syncGiftCounter(giftMessage.length);

    // Delivery scheduling — sync date input, validate stored date, render slots
    const dateInput = document.getElementById('deliveryDateInput');
    if (dateInput) {
      const minDate = getMinDate();
      dateInput.min = minDate;

      // If a saved date is now in the past (e.g. yesterday), clear it
      if (deliverySlot.date && deliverySlot.date < minDate) {
        deliverySlot.date = '';
        deliverySlot.slot = '';
        saveDelivery();
      }

      if (dateInput.value !== deliverySlot.date) dateInput.value = deliverySlot.date;
      if (deliverySlot.date) renderDeliverySlots(deliverySlot.date);
    }

    // Delivery confirmed badge
    const confirmedEl   = document.getElementById('deliveryConfirmed');
    const confirmedText = document.getElementById('deliveryConfirmedText');
    const slotsEl       = document.getElementById('deliverySlots');
    const noteEl        = document.getElementById('deliveryNote');
    if (confirmedEl) {
      if (deliverySlot.date && deliverySlot.slot) {
        const slotObj = TIME_SLOTS.find(s => s.id === deliverySlot.slot);
        confirmedText.textContent = `📦 ${formatDeliveryDate(deliverySlot.date)} · ${slotObj ? slotObj.label : ''}`;
        confirmedEl.style.display = 'flex';
        if (slotsEl) slotsEl.style.display = 'none';
        if (noteEl)  noteEl.style.display  = 'none';
      } else {
        confirmedEl.style.display = 'none';
        if (slotsEl) slotsEl.style.display = '';
        if (noteEl)  noteEl.style.display  = '';
      }
    }
  }

  function onGiftMsgInput(el) {
    giftMessage = el.value;

    syncGiftCounter(el.value.length);

  function syncGiftCounter(len) {
    const counter = document.getElementById('giftMsgCounter');
    if (!counter) return;
    counter.textContent = `${len} / 250`;
    counter.className = 'gift-msg-counter' + (len >= 250 ? ' limit' : len >= 200 ? ' warn' : '');
  }

  // ── Qty / Remove ──────────────────────────────
  function changeQty(id, delta) {
    const item = cart.find(i => i.id === id);
    if (!item) return;
    item.qty += delta;
    if (item.qty <= 0) cart = cart.filter(i => i.id !== id);
    saveCart();
    updateCartUI();
  }

  function removeItem(id) {
    cart = cart.filter(i => i.id !== id);
    saveCart();
    updateCartUI();
  }

  function toggleWish(btn) {
    btn.classList.toggle('liked');
    btn.textContent = btn.classList.contains('liked') ? '♥' : '♡';
  }

    if (!confirm('Clear your gift bag?')) return;

  function vmRenderPills(containerId, options, type, selected) {
    document.getElementById(containerId).innerHTML = options.map(opt => {
      const adj = opt.priceAdj !== 0
        ? `<span class="vm-pill-adj">${opt.priceAdj > 0 ? '+' : ''}QR ${opt.priceAdj}</span>`
        : '';
      return `<button class="vm-pill${opt.id === selected ? ' active' : ''}"

    }).join('');
  }

  function vmSelectOption(type, id, btn) {
    activeVariant[type] = id;
    btn.closest('.vm-pills').querySelectorAll('.vm-pill').forEach(b => b.classList.remove('active'));
    btn.classList.add('active');
    vmUpdatePrice();
  }

    document.getElementById('vmAddons').innerHTML = VARIANTS.addons.map(a => {
      const on = activeVariant.addons.includes(a.id);
      return `
        <button class="vm-addon${on ? ' active' : ''}" onclick="vmToggleAddon('${a.id}',this)">
          <span class="vm-addon-icon">${a.icon}</span>

  }

  function vmToggleAddon(id, btn) {
    const idx = activeVariant.addons.indexOf(id);
    if (idx === -1) { activeVariant.addons.push(id); btn.classList.add('active'); }
    else            { activeVariant.addons.splice(idx,1); btn.classList.remove('active'); }
    vmUpdatePrice();
  }

  function vmCalcPrice() {
    const sOpt = VARIANTS.sizes.find(s => s.id === activeVariant.size);
    const pOpt = VARIANTS.packaging.find(p => p.id === activeVariant.packaging);
    const addonsTotal = activeVariant.addons.reduce((sum, id) => {
      const a = VARIANTS.addons.find(x => x.id === id);
      return sum + (a ? a.price : 0);
    }, 0);
    const adj = (sOpt ? sOpt.priceAdj : 0) + (pOpt ? pOpt.priceAdj : 0);
    return Math.max(0, activeVariant.basePrice + adj) + addonsTotal;
  }

  function vmUpdatePrice() {
    document.getElementById('vmTotal').textContent = vmCalcPrice().toFixed(0);
  }

  function variantAddToCart() {
    const price   = vmCalcPrice();
    const sOpt    = VARIANTS.sizes.find(s => s.id === activeVariant.size);
    const pOpt    = VARIANTS.packaging.find(p => p.id === activeVariant.packaging);
    const addonLabels = activeVariant.addons.map(id => VARIANTS.addons.find(a => a.id === id)?.label).filter(Boolean);

    const variants = {
      size:      sOpt  ? sOpt.label  : '',
      packaging: pOpt  ? pOpt.label  : '',
      addons:    addonLabels
    };

    // Unique key: same product + same exact variant combo → increment qty
    const vKey = `${activeVariant.name}|${activeVariant.size}|${activeVariant.packaging}|${[...activeVariant.addons].sort().join(',')}`;
    const existing = cart.find(i => i.variantKey === vKey);

    if (existing) {
      existing.qty++;
    } else {
      cart.push({ id: Date.now(), name: activeVariant.name, cat: activeVariant.cat, price, basePrice: activeVariant.basePrice, qty: 1, variantKey: vKey, variants });
    }

    saveCart();
    updateCartUI();
    closeVariantModal();
    openCart();

    // Button feedback
    const addBtn = activeVariant.cardEl ? activeVariant.cardEl.querySelector('.prod-add') : null;
    if (addBtn) {
      const orig = addBtn.innerHTML;
      addBtn.innerHTML = '✓ Added'; addBtn.style.background = '#4a8c4a';
      setTimeout(() => { addBtn.innerHTML = orig; addBtn.style.background = ''; }, 1400);
    }
  }
